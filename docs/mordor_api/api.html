328    def get_threads(self, category_id: int, page: int = 1):
329        """Получить темы из раздела
330
331        Attributes:
332            category_id (int): ID категории
333            page (int): Cтраница для поиска. По умолчанию 1 (необяз.)
334            
335        Returns:
336            Словарь (dict), состоящий из списков закрепленных ('pins') и незакрепленных ('unpins') тем
337        """
338
339        soup = BeautifulSoup(self.session.get(f"{MAIN_URL}?forums/{category_id}/page-{page}").content, "html.parser")
340        result = {'pins': [], 'unpins': []}
341        for thread in soup.find_all('div', re.compile('structItem structItem--thread.*')):
342            link = thread.find_all('div', "structItem-title")[0].find_all("a")[-1]
343            if len(re.findall(r'\d+', link['href'])) < 1: continue
344            if len(thread.find_all('i', {'title': 'Закреплено'})) > 0: result['pins'].append(int(re.findall(r'\d+', urllib.parse.unquote(link['href']))[-1]))
345            else: result['unpins'].append(int(re.findall(r'\d+', urllib.parse.unquote(link['href']))[-1]))
346        return result
