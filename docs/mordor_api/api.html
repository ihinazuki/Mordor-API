 48    def get_member(self, user_id: int):
 49        """
 50        • Найти пользователя по ID •
 51        ➤ Принимает параметры:
 52            user_id - ID пользователя
 53        ➤ Возвращает массив:
 54        {
 55            "user_id": int,
 56            "username": str,
 57            "user_title": str, 
 58            "avatar": str,
 59            "messages_count": int,
 60            "reactions_count": int,
 61            "data_registration": date
 62        }
 63        """
 64
 65        content = BeautifulSoup(self.session.get(f"{MAIN_URL}?members/{user_id}").content, 'html.parser')
 66        username = content.find('h1', class_='memberHeader-name').text.strip()
 67        user_title = content.find('span', {'class': 'userTitle'}).text
 68        avatar = content.find('span', class_='avatar avatar--l avatar--default avatar--default--dynamic')
 69        if avatar:
 70            avatar = "Нету аватарки"
 71        else:
 72            avatar = content.find('span', class_='avatar avatar--l avatar--updateLink avatar--default avatar--default--dynamic')
 73            if avatar:
 74                avatar = "Нету аватарки"
 75            else:
 76                avatar = content.find('span', class_='avatar avatar--l avatar--updateLink')
 77                if avatar:
 78                    avatar = content.find('span', class_='avatar avatar--l avatar--updateLink').find('img')['src']
 79                else:
 80                    avatar = content.find('span', class_='avatar avatar--l').find('img')['src']
 81        messages_count = int(content.find('a', {'href': f'/index.php?search/member&user_id={user_id}'}).text.strip().replace(' ', ''))
 82        dl_elements = content.find_all('dl', class_='pairs pairs--rows pairs--rows--centered')
 83        data_registration = dl_elements[0].dd.get_text().replace(" ", "")
 84        reactions_count = int(dl_elements[1].dd.get_text().replace(" ", ""))
 85        return {"user_id": user_id, "username": username, "user_title": user_title, "avatar": avatar, "messages_count": messages_count, "reactions_count": reactions_count, "data_registration": data_registration}

317    def get_threads(self, category_id: int, page: int = 1):
318        """
319        • Возвращает ID всех тем из раздела •
320        ➤Принимает параметры:
321            category_id - ID раздела
322            page - номер страницы
323        ➤ Возвращает массив:
324        {
325            pins: {}
326            unpins: {}
327        }
328        """
329
330        soup = BeautifulSoup(self.session.get(f"{MAIN_URL}?forums/{category_id}/page-{page}").content, "html.parser")
331        result = {'pins': [], 'unpins': []}
332        for thread in soup.find_all('div', re.compile('structItem structItem--thread.*')):
333            link = thread.find_all('div', "structItem-title")[0].find_all("a")[-1]
334            if len(re.findall(r'\d+', link['href'])) < 1: continue
335            if len(thread.find_all('i', {'title': 'Закреплено'})) > 0: result['pins'].append(int(re.findall(r'\d+', urllib.parse.unquote(link['href']))[-1]))
336            else: result['unpins'].append(int(re.findall(r'\d+', urllib.parse.unquote(link['href']))[-1]))
337        return result
